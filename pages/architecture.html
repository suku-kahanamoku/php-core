📤 Entry point (api/products/index.php)
<?php
require_once __DIR__ . '/../../bootstrap.php';

use App\Modules\Router\{Request, Router};
use App\Modules\Database\Database;
use App\Modules\Auth\Auth;
use App\Modules\Product\ProductApi;

$request = new Request();
$router  = new Router();
$db      = Database::getInstance();
$code    = $request->franchiseCode;
$auth    = new Auth($db);

$api = new ProductApi($db, $code, $auth);
$api->registerRoutes($router);
$router->dispatch($request);

Kód vzory

*Repository – SQL vrstva
class ProductRepository {
  public function __construct(
    private Database $db,
    private string $code
  ) {}

  public function findAll(
    array $filters = []
  ): array {
    return $this->db->fetchAll(
      'SELECT * FROM product
       WHERE franchise_code = ?
       AND deleted_at IS NULL',
      [$this->code]
    );
  }

  public function findById(
    int $id
  ): ?array {
    return $this->db->fetchOne(
      'SELECT * FROM product
       WHERE id = ?
       AND franchise_code = ?',
      [$id, $this->code]
    );
  }
}
*Service – business logika
class ProductService {
  private ProductRepository $repo;

  public function __construct(
    private Database $db,
    private string $code,
    private Auth $auth
  ) {
    $this->repo =
      new ProductRepository($db, $code);
  }

  public function getAll(): array {
    // veřejný endpoint – bez auth
    return $this->repo->findAll();
  }

  public function create(
    array $data
  ): array {
    // pouze admin
    $this->auth->requireRole('admin');
    $id = $this->repo->create($data);
    return $this->repo->findById($id);
  }
}
*Api – routing
class ProductApi {
  private ProductService $service;

  public function __construct(
    private Database $db,
    private string $code,
    private Auth $auth
  ) {
    $this->service =
      new ProductService($db, $code, $auth);
  }

  public function registerRoutes(
    Router $router
  ): void {
    $router->get('/products',
      fn($req) => Response::json(
        $this->service->getAll()
      )
    );
    $router->post('/products',
      fn($req) => Response::json(
        $this->service->create($req->body),
        201
      )
    );
  }
}
💡 Jak přidat nový modul (např. Tag)
  1. Vytvořte adresář src/Modules/Tag/
  2. Přidejte tabulku do migrations/schema.sql se sloupcem franchise_code
  3. Vytvořte TagRepository.php – SQL operace
  4. Vytvořte TagService.php – business logika + auth kontroly
  5. Vytvořte TagApi.php – registrace rout přes registerRoutes()
  6. Vytvořte api/tags/index.php – entry point (zkopírujte vzor z jiného modulu)
  7. Přidejte tags do RewriteRule v api/.htaccess
  8. Přidejte testy do src/Modules/Tag/tests/test_tags.php

Login flow

Použití v Service
// Vyžaduje přihlášení (any role):
$this->auth->require();

// Vyžaduje roli admin:
$this->auth->requireRole('admin');

// Podmíněná logika:
if ($this->auth->hasRole('admin')) {
    return $this->repo->findAll();
} else {
    $userId = $this->auth->id();
    return $this->repo->findByUser($userId);
}

// Získat user ID pro filtrování:
$userId = $this->auth->id();
